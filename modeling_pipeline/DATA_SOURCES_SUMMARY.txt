SAMPLE CODE - MAKE A PREDICTION
================================================================================

from sportmonks import SportmonksAPI
import joblib
import pandas as pd

# Load your trained model
model = joblib.load('models/xgboost_model.joblib')

# Get pre-calculated features from YOUR database
from database import get_team_features

home_features = get_team_features('Liverpool')  # 438 features
away_features = get_team_features('Man City')

# Get fresh data from API
api = SportmonksAPI()
standings = api.get_standings(season_id=23614)
injuries = api.get_sidelined(fixture_id=18535517)
odds = api.get_odds(fixture_id=18535517)

# Build feature vector
features = {
    **home_features,  # Elo, form, rolling stats
    **away_features,
    'home_position': standings['Liverpool']['position'],
    'home_injuries': len(injuries['Liverpool']),
    'odds_home': odds['home'],
    # ... etc
}

X = pd.DataFrame([features])

# Predict!
probs = model.predict_proba(X)[0]
print(f"Home: {probs[2]:.1%}, Draw: {probs[1]:.1%}, Away: {probs[0]:.1%}")
